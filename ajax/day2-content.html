Cylon Eyes:

A similar example can be found here...

  • Make a new program and save it as "cylon_eyes.ino"

  • We will use the for loops above to construct a robot "looking" eye. 

                        

int timer = 100;

void setup() {
Serial.begin(9600);
// use a for loop to initialize each pin as an output:
for (int thisPin = 2; thisPin < 8; thisPin++) {
pinMode(thisPin, OUTPUT);
}
}

void loop() {
// loop from the lowest pin to the highest:
for (int thisPin = 2; thisPin < 8; thisPin++) {
// turn the pin on:
digitalWrite(thisPin, HIGH);
Serial.print(thisPin);
Serial.println();
delay(timer);
// turn the pin off:
digitalWrite(thisPin, LOW);
}

// loop from the highest pin to the lowest:
for (int thisPin = 7; thisPin >= 2; thisPin--) {
// turn the pin on:
digitalWrite(thisPin, HIGH);
Serial.print(thisPin);
Serial.println();
delay(timer);
                        
// turn the pin off:
digitalWrite(thisPin, LOW);
}
}

    We need to "map" the LED's to special digital pins on the Arduino board. Remember that if you don't connect a resistor ro the setup you can fry the LED or the board!
